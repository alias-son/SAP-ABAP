FORM modify_screen .

  LOOP AT SCREEN.
    CASE 'X'.
      WHEN p_check.
        IF screen-group1 = 'BUK'.
          screen-input = 0.
        ENDIF.
    ENDCASE.

    MODIFY SCREEN.
  ENDLOOP.

ENDFORM.
