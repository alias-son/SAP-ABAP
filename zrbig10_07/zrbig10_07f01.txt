FORM set_init_value .
  CLEAR s_bukrs.
  s_bukrs-sign = 'I'.
  s_bukrs-option = 'EQ'.
  s_bukrs-low = '4100'.
  APPEND s_bukrs.
  CLEAR s_bukrs.
ENDFORM.
*&---------------------------------------------------------------------*
*&      Form  GET_T001_DATA
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
*  -->  p1        text
*  <--  p2        text
*----------------------------------------------------------------------*
FORM get_t001_data .
  CLEAR : gt_t001, gs_t001.

  SELECT bukrs butxt ort01 waers ktopl bukrs_glob
    FROM t001
    INTO CORRESPONDING FIELDS OF TABLE gt_t001
    WHERE bukrs IN s_bukrs
    AND   ort01 IN s_ort01.

  IF sy-subrc EQ 0.
    WRITE : sy-dbcnt.
    MESSAGE S001 WITH TEXT-s01 sy-dbcnt '## #### ## #####.'.
  ELSE.

    MESSAGE s037 DISPLAY LIKE 'E'.
    STOP.
  ENDIF.
ENDFORM.
*&---------------------------------------------------------------------*
*&      Form  DISPLAY_T001_DATA
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
*  -->  p1        text
*  <--  p2        text
*----------------------------------------------------------------------*
FORM display_t001_data .
  LOOP AT gt_t001 INTO gs_t001.
    WRITE :/ gs_t001-bukrs, gs_t001-butxt, gs_t001-ort01,
             gs_t001-waers, gs_t001-ktopl, gs_t001-bukrs_glob.
  ENDLOOP.
ENDFORM.
