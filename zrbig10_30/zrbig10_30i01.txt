MODULE exit INPUT.
  CASE gv_okcode.
    WHEN 'BACK' OR 'EXIT' OR 'CANC'.
      CALL METHOD : go_alv_grid->free,
                    go_container->free.
      FREE : go_alv_grid, go_container.
      LEAVE TO SCREEN 0.
  ENDCASE.
ENDMODULE.

  ENDCASE.

ENDMODULE.
